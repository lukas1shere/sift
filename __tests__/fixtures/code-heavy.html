Authentication

The Acme API uses Bearer token authentication. All requests must include a valid API key.

Node.js

const response = await fetch('https://api.acme.dev/v1/extract', {
  method: 'POST',
  headers: {
    'Authorization': `Bearer ${process.env.ACME_API_KEY}`,
    'Content-Type': 'application/json',
  },
  body: JSON.stringify({ url: 'https://example.com' }),
});

const data = await response.json();
console.log(data.markdown);

Python

import httpx
import os

response = httpx.post(
    "https://api.acme.dev/v1/extract",
    headers={"Authorization": f"Bearer {os.environ['ACME_API_KEY']}"},
    json={"url": "https://example.com"},
)
data = response.json()
print(data["markdown"])

cURL

curl -X POST https://api.acme.dev/v1/extract \
  -H "Authorization: Bearer $ACME_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"url": "https://example.com"}'

Go

package main

import (
	"bytes"
	"encoding/json"
	"fmt"
	"net/http"
	"os"
)

func main() {
	body, _ := json.Marshal(map[string]string{"url": "https://example.com"})
	req, _ := http.NewRequest("POST", "https://api.acme.dev/v1/extract", bytes.NewBuffer(body))
	req.Header.Set("Authorization", "Bearer "+os.Getenv("ACME_API_KEY"))
	req.Header.Set("Content-Type", "application/json")

	resp, err := http.DefaultClient.Do(req)
	if err != nil {
		fmt.Println("Error:", err)
		return
	}
	defer resp.Body.Close()
	fmt.Println("Status:", resp.Status)
}

Error Handling

Always check the HTTP status code. A 401 means your API key is missing or invalid.

if (!response.ok) {
  const error = await response.json();
  throw new Error(`API error ${response.status}: ${error.message}`);
}
